# 栈

## 栈的基本概念

​	栈是一种线性数据结构，它是受限的线性表，它的特点是只能在一边加入、删除数据，我们通常把这一边叫做栈顶，另一端叫做栈底。同样的，我们也只能查询栈顶的元素，其他的元素我们默认不知道的，想要遍历访问，只能一个一个从栈顶弹出访问。

​	栈的实现方式可以用数组，也可以用链表。

## 基于数组的实现

```c++
#include <iostream>
using namespace std;

template<typename T>
class Stack {
private:
	//顺序栈的数组
	T* arr;

	//栈的大小
	int size;

	//栈的容量
	int capacity;

	void resize();

public:
	Stack() : capacity(10), arr(new T[capacity]), size(0) {};
	~Stack();
	void push(T value);
	void pop();
	T top();
	int length();
	bool empty();
};

template<typename T>
void Stack<T>::resize() {
	int newcapacity = capacity * 2;
	T* newarr = new T[newcapacity];
	for (int i = 0; i < size; i++)
	{
		newarr[i] = arr[i];
	}
	delete[] arr;
	arr = newarr;
	capacity = newcapacity;
}

//入栈
template<typename T>
void Stack<T>::push(T value) {
	//扩容
	if (size == capacity)
	{
		resize();
	}

	//正常入栈
	arr[size++] = value;
}

//出栈
template<typename T>
void Stack<T>::pop() {
	if (size == 0)
	{
		cout << "栈是空的，无法出栈！" << endl;
		exit(-1);
	}
	size--;
}

//获取栈顶元素
template<typename T>
T Stack<T>::top() {
	if (size == 0)
	{
		cout << "栈已空！" << endl;
		exit(-1);
	}

	return arr[size - 1];
}

//获取栈的深度
template<typename T>
int Stack<T>::length() {
	return size;
}

//判断栈是否为空
template<typename T>
bool Stack<T>::empty() {
	return length() == 0;
}

//栈的销毁
template<typename T>
Stack<T>::~Stack() {
	delete[] arr;
}

int main() {
	Stack<int> s;
	
	cout << s.empty() << endl;
	cout << "栈的初始大小：" << s.length() << endl;

	s.push(1);
	s.push(2);
	s.push(3);
	s.push(4);
	cout << "栈的大小：" << s.length() << endl;

	int ans = s.top();
	cout << "栈顶元素为：" << ans << endl;
	cout << "栈的大小：" << s.length() << endl;

	s.pop();
	ans = s.top();
	cout << "栈顶元素为：" << ans << endl;
	cout << "栈的大小：" << s.length() << endl;

	return 0;
}
```

## 基于链表的实现

```c++
#include <iostream>
using namespace std;

template<typename T>
class LsStack {
private:
	//链栈中的每个结点
	struct node {
		T data;
		node* next;
		node(T data) : data(data), next(nullptr) {}
	};

	//头结点指针
	node* head;

	//链栈的长度
	int size;

public:
	LsStack() : head(nullptr), size(0) {}
	~LsStack();
	void push(T value);
	void pop();
	T top();
	int length();
	bool empty();
};

//获取链栈长度
template<typename T>
int LsStack<T>::length() {
	return size;
}

//入栈，其实就是单链表的头插法
template<typename T>
void LsStack<T>::push(T value) {
	node* newnode = new node(value);
	newnode->next = head;
	head = newnode;
	++size;
}

//出栈，就是单链表删除头结点后的第一个结点
template<typename T>
void LsStack<T>::pop() {
	if (empty()) {
		cout << "栈已空！" << endl;
		exit(-1);
	}

	node* temp = head;
	node* newtop = temp->next;
	head = newtop;
	delete temp;
	--size;
}

//获取栈顶元素
template<typename T>
T LsStack<T>::top() {
	if (empty()) {
		cout << "栈已空！" << endl;
		exit(-1);
	}

	return head->data;
}

//判断栈是否为空
template<typename T>
bool LsStack<T>::empty() {
	return head == nullptr;
}

//链栈销毁
template<typename T>
LsStack<T>::~LsStack() {
	//逐个释放结点空间
	while (head != nullptr)
	{
		node* temp = head;
		head = head->next;
		delete temp;
	}
}

int main() {
	LsStack<int> lss;

	cout << lss.empty() << endl;
	cout << lss.length() << endl;

	lss.push(1);
	lss.push(2);
	cout << lss.empty() << endl;
	cout << lss.length() << endl;
	cout << lss.top() << endl;

	lss.pop();
	cout << lss.length() << endl;
	cout << lss.top() << endl;

	return 0;
}
```
